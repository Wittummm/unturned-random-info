        if (mesh.isReadable)
        {
            return true;
        }
        if ((bool)shouldLogTextureAtlasExclusions && loggedMeshes.Add(mesh))
        {
            UnturnedLog.info("Excluding mesh \"" + mesh.name + "\" in renderer \"" + renderer.GetSceneHierarchyPath() + "\" from level batching because it is not CPU-readable");
        }
        return false;
    }
